    添加新的 Agent

    第一步：创建 Agent 文件

    python 
    # backend/my_agent/my_agent.py
from typing import Annotated, TypedDict
from langchain_core.messages import BaseMessage
from langgraph.graph import StateGraph
from langgraph.graph.message import add_messages

class State(TypedDict, total=False):
    messages: Annotated[list[BaseMessage], add_messages]

def my_node(state: State):
    # 你的 Agent 逻辑
    messages = state.get("messages", [])
    return {"messages": [AIMessage(content="Hello!")]}

graph = StateGraph(State)
graph.add_node("my_node", my_node)
graph.add_edge("__start__", "my_node")
graph.add_edge("my_node", "__end__")
app = graph.compile()

    为 Agent 添加工具

    python 
    from langchain_core.tools import tool

@tool
def search_documents(query: str) -> str:
    """搜索内部文档库"""
    # 实现搜索逻辑
    return f"找到关于 {query} 的文档..."

@tool
def execute_shell_command(command: str) -> str:
    """执行 Shell 命令（训练/评估）"""
    import subprocess
    result = subprocess.run(command, shell=True, capture_output=True, text=True)
    return result.stdout or result.stderr

# 在 Agent 中使用工具
from langgraph.prebuilt import create_react_agent
app = create_react_agent(
    model=get_chat_ollama(),
    tools=[search_documents, execute_shell_command]
)

    实现人在回路

    python 
    from langgraph.types import interrupt, Command

def node_with_approval(state: State) -> Command:
    """需要用户确认才能继续的节点"""
    # 向前端发送中断，等待用户确认
    user_decision = interrupt({
        "action_request": {
            "action": "confirm_operation",
            "args": {"operation": "重要操作描述"}
        },
        "config": {
            "allow_accept": True,
            "allow_ignore": True,
        },
        "description": "是否确认执行此操作？"
    })

    if user_decision.get("action") == "confirm":
        # 执行操作
        return Command(goto="next_node", update={"status": "confirmed"})
    else:
        return Command(goto="__end__")

    添加 RAG 数据源

    在 rag/common/data_source/ 下创建新的连接器：

    python 
    # rag/common/data_source/my_connector.py
from .interfaces import BaseConnector, Document

class MyConnector(BaseConnector):
    """自定义数据源连接器"""

    def __init__(self, config: dict):
        self.api_url = config.get("api_url")
        self.api_key = config.get("api_key")

    def load_documents(self, **kwargs) -> list[Document]:
        """加载文档"""
        docs = []
        # 从数据源拉取内容
        for item in self._fetch_items():
            docs.append(Document(
                content=item["content"],
                metadata={
                    "source": self.api_url,
                    "title": item["title"],
                    "url": item["url"]
                }
            ))
        return docs

    使用 Stream Context

    typescript 
    // 在任意子组件中访问对话流
import { useStreamContext } from "@/providers/Stream";

function MyComponent() {
  const { messages, isLoading, submit, interrupt } = useStreamContext();

  const sendMessage = () => {
    submit({ messages: [{ role: "human", content: "Hello" }] });
  };

  return (
    <div>
      {isLoading && <span>AI 正在回复...</span>}
      {messages.map(msg => (
        <div key={msg.id}>{msg.content}</div>
      ))}
      <button onClick={sendMessage}>发送</button>
    </div>
  );
}

    代码规范

    Python 后端

    • 类型注解：所有公共函数参数和返回值使用类型注解
    • 文档字符串：重要函数和类添加 docstring（中英文均可）
    • 错误处理：使用 try/except 捕获异常，避免裸 raise
    • 日志：使用 print(..., flush=True) 确保日志即时输出（LangGraph Server 环境）

    TypeScript 前端

    • 所有组件使用 "use client" 或 "use server" 明确声明
    • Props 接口命名：ComponentNameProps
    • 使用 Prettier 统一代码格式（npm run format）
    • 避免 any 类型，使用 unknown + 类型守卫

    Git 提交规范

    格式

fix: 修复商用车 QA 多模态内容格式错误
